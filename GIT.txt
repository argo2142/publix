winget.exe install --id "Git.Git" --exact  --accept-source-agreements --disable-interactivity --silent --accept-package-agreements --force

winget install --id "github.cli" --silent #required to access Github through VS Code
git config --global user.name "Your Name"

Select "HTTPS (it is the default)
"Do you want to authenticate"... select "Y"
"Login with a web browser"... hit enter
Copy the ONE TIME code and hit enter to launch browser
